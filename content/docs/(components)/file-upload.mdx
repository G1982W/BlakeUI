## Usage

```tsx
import {
  FileUpload,
  FileUploadDropzone,
  FileUploadList,
  FileUploadItem,
  FileUploadItemPreview,
  FileUploadItemMetadata,
  FileUploadItemProgress,
  FileUploadItemDelete,
  FileUploadClear,
  FileUploadTrigger,
} from "@/components/ui/file-upload";
```

```tsx
<FileUpload multiple accept="image/*">
  <FileUploadDropzone />
  <FileUploadList>{/* render items for selected files */}</FileUploadList>
</FileUpload>
```
